cmake_minimum_required(VERSION 3.16)
project(vortex_installer)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -g -fPIC")
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
set(VORTEX_VERSION "1.1")
set(BUNDLE_EXECUTABLE true)

add_subdirectory(${CMAKE_CURRENT_SOURCE_DIR}/ui/installer ui_installer_build)
add_subdirectory(${CMAKE_CURRENT_SOURCE_DIR}/lib/restcpp restcpp_build)

add_executable(vortex_installer src/installer.cpp)
target_link_libraries(vortex_installer PUBLIC installer)
target_link_libraries(vortex_installer PUBLIC restclient-cpp)
target_link_libraries(vortex_installer PUBLIC spdlog)
target_link_libraries(vortex_installer PUBLIC cherry)

add_executable(vortex_update src/updater.cpp)
target_link_libraries(vortex_update PUBLIC installer)
target_link_libraries(vortex_update PUBLIC restclient-cpp)
target_link_libraries(vortex_update PUBLIC spdlog)
target_link_libraries(vortex_update PUBLIC cherry)

add_executable(vortex_uninstall src/uninstall.cpp)
target_link_libraries(vortex_uninstall PUBLIC installer)
target_link_libraries(vortex_uninstall PUBLIC restclient-cpp)
target_link_libraries(vortex_uninstall PUBLIC spdlog)
target_link_libraries(vortex_uninstall PUBLIC cherry)
